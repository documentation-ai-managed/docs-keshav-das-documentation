const cache = new VaibrixCache({
  strategy: 'intelligent',
  ttl: 3600,
  aiOptimization: true
});

app.get('/api/data', async (req, res) => {
  const key = `data_${req.params.id}`;
  let data = await cache.get(key);
  
  if (!data) {
    data = await fetchDataFromAI(req.params.id);
    await cache.set(key, data);
  }
  
  res.json(data);
});
